SELECT PRODFILHO.CODPRODUTO AS CODIGO,
       PRODFILHOCODBARRA.CODBARRA AS BARRAS,
       UNIDADE.SIGLA AS UND,
       PRODFILHO.DESCRICAO AS DESCRICAO,
       PRODFILHO.ESTQMIN AS QTD_IDEAL,
       PRODFILHO.PRECOPRATICADO2 AS PRECO_VENDA,
       PRODFILHO.ESTQATUAL AS QTD,
       PRODFILHO.PRECOFORNEC AS PRECO_CUSTO,
       PRODFILHO.CUSTOMEDIO AS CUSTO_MEDIO,
       PRODFILHO.CLASSFISCAL AS COD_NCM,
       PRODFILHO.PRDF_CST_TRIBICMS AS ST,
       PRODFILHO.CODCEST AS PERSONAL6,
       GRUPO.DESCRICAO AS GRUPO
FROM PRODFILHO
     LEFT JOIN PRODUTO ON PRODUTO.CODPRODUTO = PRODFILHO.CODPRODUTO
     LEFT JOIN GRUPO ON GRUPO.CODGRUPO = PRODUTO.CODGRUPO
     LEFT JOIN PRODFILHOCODBARRA ON PRODFILHOCODBARRA.CODPRODFILHO =
     PRODFILHO.CODPRODFILHO
     LEFT JOIN UNIDADE ON PRODFILHO.CODUNIDADE=UNIDADE.CODUNIDADE
