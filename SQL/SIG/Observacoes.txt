>> Verificar na tabela PPL.LOJPPL se usa Multilojas, e se vai pegar os dados somente de uma delas ou somar as quantidades

----SCRIPT NORMAL----
SELECT PRO.CODPRO AS CODIGO,
PRO.NOMPRO AS DESCRICAO,
PRO.RFUPRO AS COD_FABRICANTE,
PRO.EANPRO AS BARRAS,
PRO.NCMPRO AS COD_NCM,
PRO.UNDPRO AS UND,
PPL.PCOPPL AS PRECO_VENDA,
PPL.PCUPPL AS PRECO_CUSTO,
PPL.SLDPPL AS QTD,
CAT.NOMCAT AS GRUPO,
MAR.NOMMAR AS CARACTERISTICAS
FROM PRO
LEFT JOIN CAT ON PRO.CATPRO=CAT.CODCAT
LEFT JOIN PPL ON PRO.CODPRO=PPL.PROPPL
LEFT JOIN MAR ON PRO.MARPRO=MAR.CODMAR



----SCRIPT MULTILOJAS (SOMA QUANTIDADE)----
SELECT PRO.CODPRO AS CODIGO,
PRO.NOMPRO AS DESCRICAO,
PRO.RFUPRO AS COD_FABRICANTE,
PRO.EANPRO AS BARRAS,
PRO.NCMPRO AS COD_NCM,
PRO.UNDPRO AS UND,
CAT.NOMCAT AS GRUPO,
MAR.NOMMAR AS CARACTERISTICAS,
MAX (PPL.PCOPPL) AS PRECO_VENDA,
MAX (PPL.PCUPPL) AS PRECO_CUSTO,
SUM (PPL.SLDPPL) AS QTD
FROM PRO
LEFT JOIN CAT ON PRO.CATPRO=CAT.CODCAT
LEFT JOIN PPL ON PRO.CODPRO=PPL.PROPPL
LEFT JOIN MAR ON PRO.MARPRO=MAR.CODMAR 
GROUP BY 
PRO.CODPRO,
PRO.NOMPRO,
PRO.RFUPRO,
PRO.EANPRO,
PRO.NCMPRO,
PRO.UNDPRO,
CAT.NOMCAT,
MAR.NOMMAR
