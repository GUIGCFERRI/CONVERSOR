--FORNECEDORES
SELECT
  F.CODIGO
, F.RAZAO_SOCIAL AS NOME
, F.NOME_FANTASIA AS FANTASIA
, F.CNPJ AS CNPJ_CNPF
, F.IE AS IE_RG
, F.ENDERECO
, F.BAIRRO
, F.CEP
, F.TELEFONE
, F.FAX
, F.CONTATO
, F.EMAIL
, F.COMPLEMENTO
, F.NUMERO
, CI.nome AS cidade
, CI.uf
FROM FORNECEDOR F
LEFT JOIN cidade CI ON CI.codigo = F.cod_cidade
--WHERE COD_EMPRESA=1

--FORNECEDORES 2
SELECT
  C.CODIGO
, C.NOME
, C.NOME_FANTASIA AS FANTASIA
, C.CONTATO
, C.CPF_CNPJ AS CNPJ_CNPF
, C.RG_IE AS IE_RG
, CASE C.EXCLUIDO WHEN 'N' THEN 'Ativo' else 'Inativo' end AS SITUACAO
, C.CELULAR
, C.TELEFONE
--, C.OBSERVACAO
, C.OBSERVACOES
, C.EMAIL
, CI.nome AS cidade
, CI.uf
FROM cliente C
LEFT JOIN cidade CI ON CI.codigo = C.cod_cidade
WHERE C.codigo > 0
AND C.fornecedor = 'S'
--WHERE COD_EMPRESA=1
