SELECT
    PRODUTOS.CODIGO as CODIGO,
    PRODUTOS.DESCRICAO AS DESCRICAO,
    PRODUTOS.CODIGOBARRAS AS BARRAS,
    PRODUTOS.REFERENCIA AS COD_FABRICANTE,
    PRODUTOS.NCM AS COD_NCM,
    MIN(PRECOS_PROD_SERV.PRECO) AS PRECO_CUSTO,
    MAX(PRECOS_PROD_SERV.PRECO) AS PRECO_VENDA,
    PRODS_ESTOQ_IMP_OUTR.atual AS QTD,
    PRODS_ESTOQ_IMP_OUTR.margemlucro AS MARGEM_LUCRO,
    PRODUTOS.UN_COMPRAS AS UND_COMPRA,
    PRODUTOS.UN_VENDAS AS UND,
    PRODUTOS.FAT_CONVERSAO AS FAT_CONV,
    MARCAS.DESCRICAO AS CARACTERISTICAS,
    case Produtos.Ativo when 'S' then 'Ativo' else 'Inativo' end as situacao,
    PRODUTOS.CEST AS PERSONAL6
FROM PRODUTOS
LEFT JOIN MARCAS ON PRODUTOS.MARCA = MARCAS.CODIGO
LEFT JOIN PRECOS_PROD_SERV ON PRODUTOS.CODIGO = PRECOS_PROD_SERV.PRODUTO
LEFT JOIN PRODS_ESTOQ_IMP_OUTR ON PRODUTOS.CODIGO = PRODS_ESTOQ_IMP_OUTR.PRODUTO
group by 1,2,3,4,5,8,9,10,11,12,13,14,15
